# PPT智能生成与文档库管理系统 - 详细设计方案

## 1. 系统概述

### 1.1 设计目标

构建一个基于Web的PPT页面拼凑组装系统，支持用户从本地PPT文档库中检索相关页面，以"页面复制"方式组装成新PPT，做页面筛选和组合，最终结合AI 内容去修改修饰。

### 1.2 核心设计原则

- **纯复制模式**：选中页面直接复制，保留原始格式和可编辑性
- **页面级管理**：文档按页解析、向量化、检索
- **大纲先行**：先设计PPT大纲结构，再进行页面组装
- **双模式大纲**：支持智能生成和向导式生成两种大纲创建方式
- **章节驱动**：基于大纲章节结构，系统按章节检索备选页面
- **本地兼容**：允许本地化部署，数据不出境

### 3.2 PPT大纲设计模块

#### 3.2.1 功能描述

PPT大纲设计模块是页面组装的前置步骤，帮助用户快速构建清晰的PPT框架结构。系统提供两种大纲生成方式：

- **智能生成**：用户输入一段描述性文字，系统自动解析并生成完整大纲
- **向导式生成**：通过分步引导，逐步完善PPT大纲结构

**核心功能**：

- PPT主题和目标定义
- 章节结构规划
- 章节内容摘要设置
- 风格预选（可选）
- 大纲模板库
- 大纲自动保存

#### 3.2.4 智能生成算法

```python
class OutlineSmartGenerationService:
    """智能大纲生成服务"""
  
    def __init__(self, llm_client: LLMAPIClient):
        # 使用统一的LLM API客户端（支持多种云端API和自定义接口）
        self.llm = llm_client
  
    async def generate_outline(
        self,
        description: str,
        user_id: str,
        style_preference: Optional[str] = None
    ) -> SmartGenerateResponse:
        """
        根据用户描述智能生成PPT大纲
    
        Args:
            description: 用户输入的描述文字（100-2000字）
            user_id: 用户ID
            style_preference: 风格偏好（可选）
        
        Returns:
            生成的大纲和相关建议
        """
        # 1. 验证输入
        if len(description) < 100:
            raise ValueError("描述文字至少需要100字")
        if len(description) > 2000:
            description = description[:2000]
    
        # 2. 构建LLM提示词
        prompt = self._build_generation_prompt(description, style_preference)
    
        # 3. 调用LLM API生成大纲
        llm_response = await self.llm.chat_completion(prompt)
    
        # 4. 解析LLM响应
        parsed_outline = self._parse_llm_response(llm_response)
    
        # 5. 验证和补全大纲结构
        validated_outline = self._validate_and_complete(parsed_outline)
    
        # 6. 生成AI建议
        suggestions = await self._generate_suggestions(validated_outline)
    
        # 7. 保存大纲草稿
        outline = PPTOutline(
            id=generate_uuid(),
            title=validated_outline['title'],
            objective=validated_outline['objective'],
            targetAudience=validated_outline.get('target_audience', ''),
            duration=validated_outline.get('duration', ''),
            chapters=self._create_chapters(validated_outline['chapters']),
            generationType='smart',
            status='draft',
            totalPages=sum(c['page_count'] for c in validated_outline['chapters']),
            createdAt=datetime.now().isoformat(),
            updatedAt=datetime.now().isoformat(),
            userId=user_id
        )
    
        await self._save_outline(outline)
    
        return SmartGenerateResponse(
            outline=outline,
            confidence=validated_outline.get('confidence', 0.9),
            suggestions=suggestions
        )
  
    def _build_generation_prompt(
        self,
        description: str,
        style_preference: Optional[str]
    ) -> str:
        """构建智能生成提示词"""
    
        prompt = f"""你是一个专业的PPT大纲设计助手。请根据用户的描述，生成一个结构清晰、内容完整的PPT大纲。

用户描述：
{description}

{f'用户偏好风格：{style_preference}' if style_preference else ''}

请以JSON格式返回大纲结构：
{{
    "title": "PPT主标题",
    "objective": "制作目标/主要目的",
    "target_audience": "目标受众（如：外部客户、内部团队、投资人等）",
    "duration": "建议演示时长（如：15分钟）",
    "chapters": [
        {{
            "order": 1,
            "title": "章节标题",
            "page_count": 2,
            "summary": "章节内容摘要（50-100字）",
            "keywords": ["关键词1", "关键词2", "关键词3"]
        }},
        ...
    ],
    "confidence": 0.95
}}

要求：
1. 章节数量控制在3-8个之间
2. 每个章节页数建议1-5页
3. 章节内容摘要要具体、可操作
4. 关键词要有助于后续页面检索
5. 整体结构要逻辑清晰、层次分明
6. 根据内容复杂度合理分配各章节页数
"""
        return prompt
  
    def _parse_llm_response(self, response: str) -> Dict:
        """解析LLM响应"""
        try:
            # 提取JSON部分
            json_match = re.search(r'\{[\s\S]*\}', response)
            if json_match:
                return json.loads(json_match.group())
            raise ValueError("无法解析LLM响应")
        except json.JSONDecodeError as e:
            raise ValueError(f"JSON解析失败: {e}")
  
    def _validate_and_complete(self, parsed: Dict) -> Dict:
        """验证和补全大纲结构"""
        # 验证必填字段
        required_fields = ['title', 'objective', 'chapters']
        for field in required_fields:
            if field not in parsed or not parsed[field]:
                raise ValueError(f"缺少必填字段: {field}")
    
        # 验证章节数量
        chapters = parsed['chapters']
        if len(chapters) < 1:
            raise ValueError("至少需要1个章节")
        if len(chapters) > 15:
            chapters = chapters[:15]  # 最多15个章节
    
        # 补全章节信息
        for i, chapter in enumerate(chapters):
            chapter['order'] = i + 1
            chapter['page_count'] = max(1, min(5, chapter.get('page_count', 2)))
            chapter['keywords'] = chapter.get('keywords', [])[:5]  # 最多5个关键词
            if not chapter.get('summary'):
                chapter['summary'] = f"{chapter['title']}相关内容"
    
        parsed['chapters'] = chapters
        return parsed
  
    async def _generate_suggestions(self, outline: Dict) -> List[str]:
        """生成优化建议"""
        suggestions = []
    
        total_pages = sum(c['page_count'] for c in outline['chapters'])
    
        if total_pages < 5:
            suggestions.append("建议增加内容，当前页数较少")
        elif total_pages > 20:
            suggestions.append("建议精简内容，当前页数较多，可能影响演示效果")
    
        # 检查章节均衡性
        page_counts = [c['page_count'] for c in outline['chapters']]
        if max(page_counts) - min(page_counts) > 3:
            suggestions.append("建议平衡各章节页数，避免内容分布不均")
    
        # 检查关键词完整性
        chapters_without_keywords = [
            c['title'] for c in outline['chapters'] 
            if len(c.get('keywords', [])) < 2
        ]
        if chapters_without_keywords:
            suggestions.append(f"建议为以下章节补充关键词以提高检索准确性: {', '.join(chapters_without_keywords)}")
    
        return suggestions
```

**核心流程说明**：

1. **初稿生成阶段**

   - 用户只需定义章节结构和内容概要
   - 系统自动根据每个章节的描述，从文档库检索最相关的页面
   - 取Top 3-8页（用户可设置每章节页数）按相关度排序，生成章节初稿
   - 无需用户手动选择，系统智能匹配
2. **替换优化阶段**

   - 用户在预览初稿时，对不满意的页面点击"查看备选"
   - 系统基于该页内容特征，检索相似页面（3-5个备选）
   - 备选页面显示详细的源文件信息：文档名、页码、匹配度
   - 用户可选择单页或多页替换，或取消保持原页面
3. **调整完善阶段**

   - 支持跨章节拖拽调整页面顺序
   - 支持删除不满意的页面
   - 支持为某章节补充添加页面

#### 3.3.4 页面检索与初稿生成算法

```python
class SlideRetrievalService:
    def __init__(self):
        self.embedding_model = SentenceTransformer('moka-ai/m3e-base')
        self.vector_store = ChromaClient()
  
    async def generate_chapter_draft(
        self,
        chapter_description: str,
        required_pages: int = 5,
        document_ids: Optional[List[str]] = None
    ) -> List[SlideResult]:
        """
        为章节生成初稿页面
  
        根据章节描述自动检索最相关的页面，
        按匹配度排序，取前 required_pages 页作为初稿
  
        Args:
            chapter_description: 章节内容概要描述
            required_pages: 需要的页数（3-8页）
            document_ids: 限定检索范围（可选）
  
        Returns:
            初稿页面列表（已按匹配度排序）
        """
        # 限制页数范围
        required_pages = max(3, min(8, required_pages))
  
        # 1. 生成查询向量
        query_embedding = self.embedding_model.encode(chapter_description)
  
        # 2. 构建过滤条件
        filters = None
        if document_ids:
            filters = {"document_id": {"$in": document_ids}}
  
        # 3. 向量相似度检索（多取一些用于去重和过滤）
        results = self.vector_store.similarity_search(
            embedding=query_embedding,
            top_k=required_pages * 3,  # 多取3倍用于后续过滤
            filters=filters
        )
  
        # 4. 计算综合分数
        scored_results = []
        for result in results:
            vector_score = result.similarity
    
            # 结合关键词匹配分数
            text_score = self._calculate_text_similarity(
                chapter_description, 
                result.content_text
            )
    
            # 加权综合分数（70%语义 + 30%关键词）
            final_score = 0.7 * vector_score + 0.3 * text_score
    
            scored_results.append(SlideResult(
                slide_id=result.slide_id,
                document_id=result.document_id,
                document_title=result.document_title,
                page_number=result.page_number,
                thumbnail=result.thumbnail,
                similarity=round(final_score * 100, 1),  # 百分比，保留1位小数
                content_summary=result.content_text[:200]
            ))
  
        # 5. 按分数排序
        scored_results.sort(key=lambda x: x.similarity, reverse=True)
  
        # 6. 去重：同一文档的相似页面只保留最高分的
        seen_documents = set()
        unique_results = []
        for result in scored_results:
            # 使用文档ID+页码范围作为去重键（同一文档相邻页面视为相似）
            doc_key = f"{result.document_id}_{result.page_number // 2}"
            if doc_key not in seen_documents:
                seen_documents.add(doc_key)
                unique_results.append(result)
        
                # 达到需要的页数时停止
                if len(unique_results) >= required_pages:
                    break
  
        # 7. 如果结果不足，降低去重严格度再次尝试
        if len(unique_results) < required_pages:
            for result in scored_results:
                if result not in unique_results:
                    unique_results.append(result)
                    if len(unique_results) >= required_pages:
                        break
  
        return unique_results[:required_pages]
  
    async def find_alternative_slides(
        self,
        current_slide_id: str,
        top_k: int = 5
    ) -> List[SlideResult]:
        """
        查找备选页面（用于页面替换功能）
  
        基于当前页面的内容特征，检索内容相似的其他页面
        返回3-5个备选页面供用户选择，支持多选替换
  
        Args:
            current_slide_id: 当前页面ID
            top_k: 返回备选数量（默认5个，可选3-5）
  
        Returns:
            备选页面列表，按与当前页面的相似度排序
        """
        # 1. 获取当前页面的向量
        current_slide = await self.get_slide(current_slide_id)
        current_embedding = current_slide.embedding
  
        # 2. 基于向量检索相似页面
        results = self.vector_store.similarity_search(
            embedding=current_embedding,
            top_k=top_k + 5,  # 多取一些用于过滤
            exclude_ids=[current_slide_id]  # 排除当前页面
        )
  
        # 3. 格式化结果
        alternatives = []
        for result in results:
            # 计算与当前页面的相似度
            similarity = cosine_similarity(
                [current_embedding],
                [result.embedding]
            )[0][0]
    
            alternatives.append(SlideResult(
                slide_id=result.slide_id,
                document_id=result.document_id,
                document_title=result.document_title,
                page_number=result.page_number,
                thumbnail=result.thumbnail,
                similarity=round(similarity * 100, 1),
                content_summary=result.content_text[:200]
            ))
  
        # 4. 按相似度排序
        alternatives.sort(key=lambda x: x.similarity, reverse=True)
  
        return alternatives[:top_k]
  
    async def search_slides(
        self, 
        query: str, 
        top_k: int = 20,
        filters: Optional[Dict] = None
    ) -> List[SlideResult]:
        """
        通用检索接口
  
        用于手动搜索页面（补充添加页面时使用）
        """
        query_embedding = self.embedding_model.encode(query)
  
        results = self.vector_store.similarity_search(
            embedding=query_embedding,
            top_k=top_k,
            filters=filters
        )
  
        return self._format_results(results)
```

```typescript
// 操作记录
interface Operation {
  id: string;
  type: OperationType;
  timestamp: number;
  // 操作前状态（用于撤销）
  prevState: any;
  // 操作后状态（用于重做）
  nextState: any;
  // 操作描述（用于显示）
  description: string;
  // 操作涉及的章节/页面ID（用于快速定位）
  chapterId?: string;
  pageId?: string;
}

// 撤销管理器
class UndoManager {
  private undoStack: Operation[] = [];
  private redoStack: Operation[] = [];
  private maxHistory = 20;  // 最大保存20步
  
  // 执行操作
  execute(operation: Operation): void {
    // 执行操作
    this.applyOperation(operation);
  
    // 添加到撤销栈
    this.undoStack.push(operation);
    if (this.undoStack.length > this.maxHistory) {
      this.undoStack.shift();
    }
  
    // 清空重做栈
    this.redoStack = [];
  
    // 自动保存草稿
    this.autoSave();
  }
  
  // 撤销
  undo(): boolean {
    if (this.undoStack.length === 0) return false;
  
    const operation = this.undoStack.pop()!;
  
    // 恢复到之前的状态
    this.restoreState(operation.prevState);
  
    // 添加到重做栈
    this.redoStack.push(operation);
  
    return true;
  }
  
  // 重做
  redo(): boolean {
    if (this.redoStack.length === 0) return false;
  
    const operation = this.redoStack.pop()!;
  
    // 恢复到之后的状态
    this.restoreState(operation.nextState);
  
    // 添加回撤销栈
    this.undoStack.push(operation);
  
    return true;
  }
  
  canUndo(): boolean {
    return this.undoStack.length > 0;
  }
  
  canRedo(): boolean {
    return this.redoStack.length > 0;
  }
  
  // 获取操作描述（用于UI显示）
  getUndoDescription(): string | null {
    if (this.undoStack.length === 0) return null;
    return this.undoStack[this.undoStack.length - 1].description;
  }
  
  getRedoDescription(): string | null {
    if (this.redoStack.length === 0) return null;
    return this.redoStack[this.redoStack.length - 1].description;
  }
}
```

#### 3.4.1 功能描述

#### 3.4.2 导出流程

```python
class PPTExportService:
    def __init__(self):
        self.template_prs = Presentation()  # 新建PPT
  
    async def export_assembly(
        self,
        draft_id: str,
        output_path: str
    ) -> str:
        """
        导出组装好的PPT
  
        使用python-pptx的页面复制功能，
        保留所有原始格式和可编辑性
        """
        # 1. 获取草稿信息
        draft = await self.get_draft(draft_id)
  
        # 2. 创建新PPT
        new_prs = Presentation()
  
        # 3. 复制选中的页面
        slide_index = 0
        for chapter in draft.chapters:
            for selected in chapter.selected_pages:
                # 获取源页面
                source_slide = await self.get_slide_from_document(
                    selected.document_id,
                    selected.page_number
                )
      
                # 复制页面到新PPT
                self._copy_slide(
                    source_prs=source_slide.presentation,
                    source_slide_idx=source_slide.page_number - 1,
                    target_prs=new_prs,
                    target_slide_idx=slide_index
                )
      
                slide_index += 1
  
        # 4. 保存文件
        new_prs.save(output_path)
  
        return output_path
  
    def _copy_slide(
        self,
        source_prs: Presentation,
        source_slide_idx: int,
        target_prs: Presentation,
        target_slide_idx: int
    ):
        """
        复制幻灯片，保留所有元素和格式
  
        python-pptx本身不支持直接复制slide，
        需要手动复制所有shapes
        """
        source_slide = source_prs.slides[source_slide_idx]
  
        # 使用空白布局创建新slide
        blank_layout = target_prs.slide_layouts[6]  # 空白布局
        new_slide = target_prs.slides.add_slide(blank_layout)
  
        # 复制背景
        if source_slide.background.fill.type:
            self._copy_background(source_slide, new_slide)
  
        # 复制所有shapes
        for shape in source_slide.shapes:
            self._copy_shape(shape, new_slide)
  
        return new_slide
  
    def _copy_shape(self, source_shape, target_slide):
        """复制单个shape到新slide"""
  
        # 根据shape类型分别处理
        if source_shape.has_text_frame:
            # 文本框 - 复制文本和格式
            new_shape = target_slide.shapes.add_textbox(
                source_shape.left,
                source_shape.top,
                source_shape.width,
                source_shape.height
            )
            self._copy_text_frame(source_shape.text_frame, new_shape.text_frame)
  
        elif source_shape.has_table:
            # 表格 - 复制表格结构和内容
            new_table = target_slide.shapes.add_table(
                rows=source_shape.table.rows,
                cols=source_shape.table.columns,
                left=source_shape.left,
                top=source_shape.top,
                width=source_shape.width,
                height=source_shape.height
            )
            self._copy_table(source_shape.table, new_table.table)
  
        elif source_shape.has_chart:
            # 图表 - 复制图表数据和类型
            # 图表复制较复杂，需要重新创建
            self._copy_chart(source_shape.chart, target_slide)
  
        elif source_shape.shape_type == MSO_SHAPE_TYPE.PICTURE:
            # 图片 - 复制图片数据
            image_bytes = source_shape.image.blob
            new_shape = target_slide.shapes.add_picture(
                image_bytes,
                source_shape.left,
                source_shape.top,
                source_shape.width,
                source_shape.height
            )
```

### 3.6 PPT精修模块（AI对话编辑）

#### 3.6.1 功能描述

PPT精修模块是在组装完成基础上，提供基于自然语言对话的AI辅助编辑功能。用户选中特定页面后，通过对话方式指示AI修改页面内容，包括：

- **样式调整**：修改字体、颜色、对齐方式
- **元素增删**：添加新元素或删除现有元素
- **布局优化**：调整元素位置和大小

**核心特点**：

- 对话式交互，自然语言指令
- 所见即所得，实时预览修改效果
- 支持撤销/重做每次修改
- 保留原始格式和样式一致性

```python
class PPTRefinementService:
    def __init__(self, llm_client: LLMAPIClient):
        # 使用统一的LLM API客户端
        self.llm = llm_client
        self.ppt_editor = SlideEditor()
  
    async def process_user_command(
        self,
        task_id: str,
        page_index: int,
        user_message: str,
        conversation_history: List[Message]
    ) -> CommandResult:
        """
        处理用户对话指令
  
        1. 理解用户意图
        2. 定位目标元素
        3. 执行修改操作
        4. 返回修改结果
        """
        # 1. 获取当前页面内容
        current_page = await self.get_page_content(task_id, page_index)
  
        # 2. 构建LLM提示词
        prompt = self._build_refinement_prompt(
            page_content=current_page,
            user_message=user_message,
            history=conversation_history
        )
  
        # 3. 调用LLM API解析意图
        llm_response = await self.llm.chat_completion(prompt)
  
        # 4. 解析LLM返回的修改指令
        modification_plan = self._parse_modification_plan(llm_response)
  
        # 5. 执行修改
        if modification_plan.confidence > 0.7:
            result = await self._execute_modification(
                task_id=task_id,
                page_index=page_index,
                plan=modification_plan
            )
      
            # 6. 生成回复消息
            response_message = self._generate_response(
                plan=modification_plan,
                result=result
            )
      
            return CommandResult(
                success=True,
                message=response_message,
                modification=result.modification_record,
                updated_page=result.updated_page
            )
        else:
            # 置信度低，请求用户澄清
            return CommandResult(
                success=False,
                message=self._generate_clarification_request(modification_plan),
                needs_clarification=True
            )
  
    def _build_refinement_prompt(
        self,
        page_content: SlideContent,
        user_message: str,
        history: List[Message]
    ) -> str:
        """构建精修提示词"""
  
        # 提取页面元素描述
        elements_desc = self._describe_elements(page_content.elements)
  
        prompt = f"""你是一个PPT编辑助手。请分析用户的编辑指令，并生成具体的修改计划。

当前页面的元素：
{elements_desc}

用户指令："{user_message}"

请分析用户的意图，并以JSON格式返回：
{{
    "intent": "修改意图类型",
    "target_element": "目标元素ID（如有）",
    "target_type": "text/image/table/chart",
    "modification": {{
        "action": "具体动作",
        "parameters": {{ 动作参数 }}
    }},
    "description": "对修改的简短描述",
    "confidence": 0.95  // 置信度0-1
}}

支持的意图类型：
- edit_text: 编辑文本内容
- edit_table: 编辑表格
- replace_image: 替换图片
- add_element: 添加元素
- delete_element: 删除元素
- move_element: 移动元素
- change_style: 修改样式（颜色、字体等）
- generate_content: 根据描述生成内容

如果指令不明确，请返回：
{{
    "intent": "unknown",
    "clarification_questions": ["问题1", "问题2"],
    "confidence": 0.3

        return prompt
  
    async def _execute_modification(
        self,
        task_id: str,
        page_index: int,
        plan: ModificationPlan
    ) -> ModificationResult:
        """执行具体的修改操作"""
  
        action = plan.modification['action']
        params = plan.modification['parameters']
  
        if plan.target_type == 'text':
            return await self._modify_text(
                task_id, page_index, 
                plan.target_element, action, params
            )
        elif plan.target_type == 'table':
            return await self._modify_table(
                task_id, page_index,
                plan.target_element, action, params
            )
        elif plan.target_type == 'image':
            return await self._modify_image(
                task_id, page_index,
                plan.target_element, action, params
            )
        # ... 其他类型
  
    async def _modify_text(
        self,
        task_id: str,
        page_index: int,
        element_id: str,
        action: str,
        params: Dict
    ) -> ModificationResult:
        """修改文本元素"""
  
        # 获取当前文本内容
        current = await self.ppt_editor.get_element(task_id, page_index, element_id)
  
        # 根据动作执行修改
        if action == 'replace':
            new_text = params.get('new_text', '')
            updated = await self.ppt_editor.set_text(
                task_id, page_index, element_id, new_text
            )
        elif action == 'append':
            appended_text = params.get('text', '')
            new_text = current.text + appended_text
            updated = await self.ppt_editor.set_text(
                task_id, page_index, element_id, new_text
            )
        elif action == 'edit_style':
            style_changes = params.get('style', {})
            updated = await self.ppt_editor.set_style(
                task_id, page_index, element_id, style_changes
            )
  
        # 生成修改记录
        modification = ModificationRecord(
            type=ModificationType.EDIT_TEXT,
            element_id=element_id,
            before=current,
            after=updated
        )
  
        return ModificationResult(
            success=True,
            modification_record=modification,
            updated_page=await self.get_page_content(task_id, page_index)
        )
  
    async def generate_content_with_ai(
        self,
        task_id: str,
        page_index: int,
        context: str,
        content_type: str
    ) -> GeneratedContent:
        """
        使用AI生成内容
  
        例如：根据上下文生成标题、补充段落、优化表述等
        """
        page_content = await self.get_page_content(task_id, page_index)
  
        prompt = f"""请为PPT页面生成{content_type}内容。

页面当前内容：
{self._extract_page_text(page_content)}

生成要求：{context}

要求：
1. 内容简洁专业，适合PPT展示
2. 保持与现有内容的风格一致
3. 返回纯文本内容
"""
  
        generated = await self.llm.chat_completion(prompt)
  
        return GeneratedContent(
            content=generated.strip(),
            type=content_type,
            confidence=0.9
        )
```

```python
class RefinementSuggestionService:

        """分析文本质量，提供优化建议"""
        suggestions = []
  
        for element in page.elements:
            if element.type == 'text':
                text = element.content.text
          
                # 检测过长的段落
                if len(text) > 100:
                    suggestions.append(Suggestion(
                        type='text_length',
                        description='文本内容较长，建议精简',
                        element_id=element.id,
                        action='shorten',
                        preview=self._generate_shortened_text(text)
                    ))
          
                # 检测口语化表达
                if self._detect_colloquial(text):
                    suggestions.append(Suggestion(
                        type='tone',
                        description='文本包含口语化表达，建议改为书面语',
                        element_id=element.id,
                        action='formalize'
                    ))
  
        return suggestions
  
    async def get_quick_actions(
        self,

    ) -> List[QuickAction]:
        """
```

```python
                             title=draft.title, status='editing',
                             total_pages=draft.total_pages)

        for idx, page in enumerate(draft.all_pages):
            content = await self._parse_slide_to_json(page)
            refined_page = RefinedPage(
                task_id=task.id, page_index=idx,
                content=content, version=1)
            self.db.add(refined_page)
      
        self.db.commit()
        return task

                          message: str, element_id: str = None):
        """处理用户对话并执行修改"""
        page = await self._get_page(task_id, page_index)
      
        # 构建提示词并调用LLM
        prompt = self._build_prompt(page.content, message, element_id)

            mod = await self._execute_modification(page, plan)
            return {"success": True, "modification": mod}
```

```python
"""
LLM API客户端 - 统一的大模型接口层

设计原则：
1. 仅支持API接口模式，不支持私有化部署
2. 支持多种云端LLM提供商（OpenAI、Claude、通义千问、混元等）
3. 支持自定义API接口，满足企业内部模型接入需求
4. 统一的接口抽象，便于切换不同模型
"""

from abc import ABC, abstractmethod
from typing import List, Dict, Optional, Any
import httpx
from pydantic import BaseModel


class LLMMessage(BaseModel):
    """LLM消息格式"""
    role: str  # system, user, assistant
    content: str


class LLMResponse(BaseModel):
    """LLM响应格式"""
    content: str
    model: str
    usage: Optional[Dict[str, int]] = None
    finish_reason: Optional[str] = None


class BaseLLMProvider(ABC):
    """LLM提供商抽象基类"""
  
    @abstractmethod
    async def chat_completion(
        self,
        messages: List[LLMMessage],
        temperature: float = 0.7,
        max_tokens: int = 2048,
        **kwargs
    ) -> LLMResponse:
        """调用LLM完成对话"""
        pass


class OpenAIProvider(BaseLLMProvider):
    """OpenAI API提供商"""
  
    def __init__(self, api_key: str, model: str = "gpt-4o", base_url: str = None):
        self.api_key = api_key
        self.model = model
        self.base_url = base_url or "https://api.openai.com/v1"
  
    async def chat_completion(
        self,
        messages: List[LLMMessage],
        temperature: float = 0.7,
        max_tokens: int = 2048,
        **kwargs
    ) -> LLMResponse:
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                f"{self.base_url}/chat/completions",
                headers={"Authorization": f"Bearer {self.api_key}"},
                json={
                    "model": self.model,
                    "messages": [m.dict() for m in messages],
                    "temperature": temperature,
                    "max_tokens": max_tokens,
                    **kwargs
                },
                timeout=120
            )
            resp.raise_for_status()
            data = resp.json()
            return LLMResponse(
                content=data["choices"][0]["message"]["content"],
                model=data["model"],
                usage=data.get("usage"),
                finish_reason=data["choices"][0].get("finish_reason")
            )

    """Anthropic Claude API提供商"""
  
    def __init__(self, api_key: str, model: str = "claude-3-5-sonnet-20241022"):
        self.api_key = api_key
        self.model = model

    async def chat_completion(
        self,
        messages: List[LLMMessage],
        temperature: float = 0.7,
        max_tokens: int = 2048,
        **kwargs
    ) -> LLMResponse:
        # 转换消息格式适配Claude API
        system_msg = None
        chat_messages = []
        for msg in messages:
            if msg.role == "system":
                system_msg = msg.content
            else:
                chat_messages.append({"role": msg.role, "content": msg.content})
    
        async with httpx.AsyncClient() as client:
            request_body = {
                "model": self.model,
                "messages": chat_messages,
                "max_tokens": max_tokens,
                "temperature": temperature,
            }
            if system_msg:
                request_body["system"] = system_msg
        
            resp = await client.post(
                f"{self.base_url}/messages",
                headers={
                    "x-api-key": self.api_key,
                    "anthropic-version": "2023-06-01",
                    "content-type": "application/json"
                },
                json=request_body,
                timeout=120
            )
            resp.raise_for_status()
            data = resp.json()
            return LLMResponse(
                content=data["content"][0]["text"],
                model=data["model"],
                usage=data.get("usage"),
                finish_reason=data.get("stop_reason")
            )

class CustomAPIProvider(BaseLLMProvider):
    """自定义API提供商 - 支持用户自定义接口"""
  
    def __init__(
        self,
        base_url: str,
        api_key: str = None,
        model: str = "default",
        headers: Dict[str, str] = None,
        request_template: Dict[str, Any] = None,
        response_path: str = "choices.0.message.content"
    ):
        """
        初始化自定义API提供商
    
        Args:
            base_url: API基础URL
            api_key: API密钥（可选）
            model: 模型名称
            headers: 自定义请求头
            request_template: 请求体模板，支持变量替换
            response_path: 响应内容提取路径（点分隔）
        """
        self.base_url = base_url.rstrip('/')
        self.api_key = api_key
        self.model = model
        self.custom_headers = headers or {}
        self.request_template = request_template
        self.response_path = response_path
  
    async def chat_completion(
        self,
        messages: List[LLMMessage],
        temperature: float = 0.7,
        max_tokens: int = 2048,
        **kwargs
    ) -> LLMResponse:
        headers = {"Content-Type": "application/json"}
        if self.api_key:
            headers["Authorization"] = f"Bearer {self.api_key}"
        headers.update(self.custom_headers)
    
        # 构建请求体
        if self.request_template:
            request_body = self._build_request_from_template(
                messages, temperature, max_tokens, kwargs
            )
        else:
            # 默认OpenAI兼容格式
            request_body = {
                "model": self.model,
                "messages": [m.dict() for m in messages],
                "temperature": temperature,
                "max_tokens": max_tokens,
            }
    
        async with httpx.AsyncClient() as client:
            resp = await client.post(
                f"{self.base_url}/chat/completions",
                headers=headers,
                json=request_body,
                timeout=120
            )
            resp.raise_for_status()
            data = resp.json()
        
            # 从响应中提取内容
            content = self._extract_content(data, self.response_path)
        
            return LLMResponse(
                content=content,
                model=self.model,
                usage=data.get("usage")
            )
  
    def _extract_content(self, data: dict, path: str) -> str:
        """根据路径从响应中提取内容"""
        keys = path.split('.')
        result = data
        for key in keys:
            if key.isdigit():
                result = result[int(key)]
            else:
                result = result[key]
        return result
  
    def _build_request_from_template(
        self,
        messages: List[LLMMessage],
        temperature: float,
        max_tokens: int,
        kwargs: dict
    ) -> dict:
        """根据模板构建请求体"""
        import copy
        body = copy.deepcopy(self.request_template)
        # 替换变量
        def replace_vars(obj):
            if isinstance(obj, str):
                obj = obj.replace("{{model}}", self.model)
                obj = obj.replace("{{temperature}}", str(temperature))
                obj = obj.replace("{{max_tokens}}", str(max_tokens))
                return obj
            elif isinstance(obj, dict):
                return {k: replace_vars(v) for k, v in obj.items()}
            elif isinstance(obj, list):
                return [replace_vars(v) for v in obj]
            return obj
    
        body = replace_vars(body)
        body["messages"] = [m.dict() for m in messages]
        return body


class LLMAPIClient:
    """
    统一LLM API客户端
  
    使用示例:
        # 使用预置提供商
        client = LLMAPIClient(provider="openai", api_key="sk-xxx")
    
        # 使用自定义API
        client = LLMAPIClient(
            provider="custom",
            custom_config={
                "base_url": "https://your-api.com/v1",
                "api_key": "your-key",
                "model": "your-model"
            }
        )
    
        response = await client.chat_completion("你好")
    """
  
    PROVIDERS = {
        "openai": OpenAIProvider,
        "claude": ClaudeProvider,
        "qwen": QwenProvider,
        "hunyuan": HunyuanProvider,
        "custom": CustomAPIProvider,
    }
  
    def __init__(
        self,
        provider: str = None,
        api_key: str = None,
        model: str = None,
        custom_config: Dict[str, Any] = None,
        **kwargs
    ):
        """
        初始化LLM API客户端
    
        Args:
            provider: 提供商名称 (openai, claude, qwen, hunyuan, custom)
            api_key: API密钥
            model: 模型名称
            custom_config: 自定义API配置（当provider为custom时使用）
        """
        # 从配置读取默认值
        provider = provider or settings.LLM_PROVIDER
        api_key = api_key or settings.LLM_API_KEY
        model = model or settings.LLM_MODEL
    
        if provider not in self.PROVIDERS:
            raise ValueError(f"不支持的LLM提供商: {provider}")
    
        if provider == "custom":
            config = custom_config or settings.LLM_CUSTOM_CONFIG or {}
            self.provider_instance = CustomAPIProvider(**config)
        elif provider == "hunyuan":
            self.provider_instance = HunyuanProvider(
                secret_id=settings.HUNYUAN_SECRET_ID,
                secret_key=settings.HUNYUAN_SECRET_KEY,
                model=model
            )
        else:
            provider_class = self.PROVIDERS[provider]
            self.provider_instance = provider_class(api_key=api_key, model=model, **kwargs)
  
    async def chat_completion(
        self,
        prompt: str,
        system_prompt: str = None,
        temperature: float = 0.7,
        max_tokens: int = 2048,
        **kwargs
    ) -> str:
        """
        简化的对话接口
    
        Args:
            prompt: 用户输入
            system_prompt: 系统提示词（可选）
            temperature: 温度参数
            max_tokens: 最大token数
        
        Returns:
            模型响应文本
        """
        messages = []
        if system_prompt:
            messages.append(LLMMessage(role="system", content=system_prompt))
        messages.append(LLMMessage(role="user", content=prompt))
    
        response = await self.provider_instance.chat_completion(
            messages=messages,
            temperature=temperature,
            max_tokens=max_tokens,
            **kwargs
        )
        return response.content
  
    async def chat_with_history(
        self,
        messages: List[Dict[str, str]],
        temperature: float = 0.7,
        max_tokens: int = 2048,
        **kwargs
    ) -> str:
        """
        带历史记录的对话接口
    
        Args:
            messages: 消息列表 [{"role": "user", "content": "..."}]
            temperature: 温度参数
            max_tokens: 最大token数
        
        Returns:
            模型响应文本
        """
        llm_messages = [LLMMessage(**msg) for msg in messages]
        response = await self.provider_instance.chat_completion(
            messages=llm_messages,
            temperature=temperature,
            max_tokens=max_tokens,
            **kwargs
        )
        return response.content
```

```python
        element['content']['text'] = new_text
    
        # 应用样式修改
        if style_changes:
            element['style'].update(style_changes)
    
        return {"before": old_text, "after": new_text}
  
    async def modify_table(self, content: dict, element_id: str,
                          operation: str, data: dict):
        """修改表格"""
        element = self._find_element(content, element_id)
        table = element['content']
    
        if operation == 'update_cell':
            row, col = data['row'], data['col']
            table['cells'][row][col]['text'] = data['value']
        elif operation == 'add_row':
            table['rows'] += 1
            table['cells'].append([{"text": ""} for _ in range(table['cols'])])
        elif operation == 'delete_row':
            table['rows'] -= 1
            table['cells'].pop(data['row'])
    
        return element
```

```python
class WebSearchService:
    """联网搜索服务"""
  
    def __init__(self, llm_client: LLMAPIClient):
        """
        初始化联网搜索服务
      
        Args:
            llm_client: LLM API客户端实例，用于智能关键词提取
        """
        self.llm = llm_client
        self.search_providers = [
            BingSearchProvider(),
            GoogleSearchProvider(),
        ]
        self.content_extractor = ContentExtractor()
  
    async def search_topic(
        self,
        topic: str,
        max_results: int = 15,
        language: str = "zh-CN"
    ) -> List[WebSource]:
        """
        根据主题进行联网搜索
      
        1. 从主题中提取搜索关键词
        2. 并行调用多个搜索引擎
        3. 合并去重搜索结果
        4. 按相关度排序
        5. 提取正文内容摘要
        """
        # 提取关键词
        keywords = await self._extract_keywords(topic)
      
        # 并行搜索
        all_results = []
        tasks = [
            provider.search(keywords, max_results)
            for provider in self.search_providers
        ]
        results_list = await asyncio.gather(*tasks, return_exceptions=True)
      
        for results in results_list:
            if isinstance(results, list):
                all_results.extend(results)
      
        # 去重和排序
        unique_results = self._deduplicate(all_results)
        ranked_results = self._rank_by_relevance(unique_results, topic)
      
        # 提取摘要
        for result in ranked_results[:max_results]:
            result.snippet = await self.content_extractor.extract_summary(
                result.url, max_length=500
            )
      
        return ranked_results[:max_results]
  
    async def _extract_keywords(self, topic: str) -> List[str]:
        """使用AI提取搜索关键词"""
        prompt = f"""
        请从以下主题描述中提取3-5个最重要的搜索关键词：
      
        主题：{topic}
      
        只返回关键词，用逗号分隔。
        """
        response = await self.llm.chat_completion(prompt)
        return [kw.strip() for kw in response.split(',')]
```

```python
class PPTGenerationService:
    """PPT内容生成服务"""
    
    def __init__(self, llm_client: LLMAPIClient):
        """
        初始化PPT内容生成服务
        
        Args:
            llm_client: LLM API客户端实例，用于内容生成
        """
        self.llm = llm_client
        self.search_service = WebSearchService(llm_client)
        self.template_service = TemplateService()
    
    async def generate_ppt(
        self,
        task: GenerationTask,
        progress_callback: Callable[[int, str], None]
    ) -> GeneratedPPT:
        """
        完整的PPT生成流程
        """
        try:
            # 阶段1: 联网搜索
            progress_callback(10, "正在搜索相关信息...")
            web_sources = await self.search_service.search_topic(task.topic)
            task.webSources = web_sources
            
            # 阶段2: 内容分析
            progress_callback(30, "正在分析搜索结果...")
            analysis = await self._analyze_content(task.topic, web_sources)
            
            # 阶段3: 生成大纲
            progress_callback(40, "正在生成PPT大纲...")
            outline = await self._generate_outline(
                task.topic, 
                task.pageCount, 
                task.pptType,
                analysis
            )
            task.generatedOutline = outline
            
            # 阶段4: 逐页生成内容
            slides = []
            for i, section in enumerate(outline.sections):
                progress = 40 + int((i + 1) / len(outline.sections) * 40)
                progress_callback(progress, f"正在生成第 {i+1}/{len(outline.sections)} 页...")
                
                slide = await self._generate_slide_content(
                    section, 
                    task.pptType,
                    web_sources
                )
                slides.append(slide)
            
            task.generatedSlides = slides
            
            # 阶段5: 应用风格模板
            progress_callback(85, "正在应用风格模板...")
            template = await self.template_service.get_template(task.templateId)
            styled_slides = await self._apply_template_style(slides, template)
            
            progress_callback(100, "生成完成")
            
            return GeneratedPPT(
                task_id=task.id,
                slides=styled_slides,
                template=template,
                sources=web_sources
            )
            
        except Exception as e:
            task.status = 'failed'
            task.error = str(e)
            raise
    
    async def _analyze_content(
        self,
        topic: str,
        sources: List[WebSource]
    ) -> ContentAnalysis:
        """分析搜索内容，提取关键信息"""
        sources_text = "\n".join([
            f"- {s.title}: {s.snippet}" for s in sources[:10]
        ])
        
        prompt = f"""
        请分析以下主题相关的搜索结果，提取关键信息：
        
        主题：{topic}
        
        搜索结果：
        {sources_text}
        
        请提供：
        1. 内容摘要（200字以内）
        2. 关键要点（5-8个）
        3. 建议的PPT结构
        
        请以JSON格式返回。
        """
        response = await self.llm.chat_completion(prompt, response_format="json")
        return ContentAnalysis.from_json(response)
    
    async def _generate_outline(
        self,
        topic: str,
        page_count: int,
        ppt_type: str,
        analysis: ContentAnalysis
    ) -> Outline:
        """生成PPT大纲"""
        prompt = f"""
        请为以下主题生成一个{page_count}页的PPT大纲：
        
        主题：{topic}
        PPT类型：{ppt_type}
        
        参考资料摘要：
        {analysis.summary}
        
        要求：
        1. 第一页为封面页
        2. 第二页为目录/概览
        3. 最后一页为总结/联系方式
        4. 中间页面按逻辑顺序组织内容
        5. 每页包含标题和3-5个要点
        
        请以JSON格式返回大纲结构。
        """
        response = await self.llm.chat_completion(prompt, response_format="json")
        return Outline.from_json(response)
    
    async def _generate_slide_content(
        self,
        section: OutlineSection,
        ppt_type: str,
        sources: List[WebSource]
    ) -> SlideContent:
        """为单页生成详细内容"""
        # 找到与该章节最相关的来源
        relevant_sources = self._find_relevant_sources(section, sources)
        
        prompt = f"""
        请为PPT的以下章节生成详细内容：
        
        章节标题：{section.title}
        要点：{', '.join(section.points)}
        
        参考资料：
        {self._format_sources(relevant_sources)}
        
        要求：
        1. 生成一个吸引人的页面标题
        2. 为每个要点扩展成2-3句话
        3. 如果适合，建议添加图表类型
        4. 标注内容来源
        
        请以JSON格式返回。
        """
        response = await self.llm.chat_completion(prompt, response_format="json")
        return SlideContent.from_json(response)
    
    def _find_relevant_sources(
        self,
        section: OutlineSection,
        sources: List[WebSource]
    ) -> List[WebSource]:
        """找到与章节最相关的来源"""
        # 基于标题和要点关键词匹配
        keywords = [section.title] + section.points
        scored_sources = []
        
        for source in sources:
            score = sum(
                1 for kw in keywords 
                if kw.lower() in source.title.lower() or 
                   kw.lower() in (source.snippet or '').lower()
            )
            scored_sources.append((score, source))
        
        scored_sources.sort(key=lambda x: x[0], reverse=True)
        return [s for _, s in scored_sources[:3]]
    
    def _format_sources(self, sources: List[WebSource]) -> str:
        """格式化来源为文本"""
        return "\n".join([
            f"- [{s.title}]({s.url}): {s.snippet}" 
            for s in sources
        ])
```

### 4.2 表结构定义

```sql
-- 用户表
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    username VARCHAR(50) UNIQUE NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    is_active BOOLEAN DEFAULT TRUE
);

-- 文档分类表
CREATE TABLE categories (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(user_id, name)
);

-- 文档表
CREATE TABLE documents (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    category_id UUID REFERENCES categories(id) ON DELETE SET NULL,
    title VARCHAR(255) NOT NULL,
    file_path VARCHAR(500) NOT NULL,
    file_size BIGINT NOT NULL,
    page_count INTEGER NOT NULL DEFAULT 0,
    status VARCHAR(20) DEFAULT 'parsing', -- parsing, ready, error
    error_message TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- PPT页面表
CREATE TABLE slides (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    document_id UUID REFERENCES documents(id) ON DELETE CASCADE,
    page_number INTEGER NOT NULL,
    title VARCHAR(500),
    content_text TEXT,              -- 纯文本内容（用于向量化）
    layout_type VARCHAR(50),
    elements JSONB,                 -- 元素结构（文本、图片等）
    thumbnail_path VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(document_id, page_number)
);

-- 组装草稿表
CREATE TABLE assembly_drafts (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    description TEXT,
    status VARCHAR(20) DEFAULT 'draft', -- draft, completed, exported
    version INTEGER DEFAULT 1,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 草稿章节表
CREATE TABLE draft_chapters (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    draft_id UUID REFERENCES assembly_drafts(id) ON DELETE CASCADE,
    title VARCHAR(255) NOT NULL,
    description TEXT,               -- 内容概要（用于检索）
    required_pages INTEGER NOT NULL DEFAULT 1,
    order_index INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 章节已选页面表
CREATE TABLE chapter_selected_pages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    chapter_id UUID REFERENCES draft_chapters(id) ON DELETE CASCADE,
    slide_id UUID REFERENCES slides(id) ON DELETE CASCADE,
    document_id UUID REFERENCES documents(id) ON DELETE CASCADE,
    page_number INTEGER NOT NULL,
    order_index INTEGER NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(chapter_id, slide_id)
);

-- 操作历史表（用于撤销/重做）
CREATE TABLE operation_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    draft_id UUID REFERENCES assembly_drafts(id) ON DELETE CASCADE,
    operation_type VARCHAR(50) NOT NULL,
    operation_data JSONB NOT NULL,  -- 操作数据
    prev_state JSONB,               -- 操作前状态
    description VARCHAR(255),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 导出记录表
CREATE TABLE export_records (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    draft_id UUID REFERENCES assembly_drafts(id) ON DELETE SET NULL,
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    file_path VARCHAR(500) NOT NULL,
    file_size BIGINT,
    page_count INTEGER,
    exported_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 风格模板表
CREATE TABLE style_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    source_document_id UUID REFERENCES documents(id) ON DELETE SET NULL,
    color_scheme JSONB,             -- 配色方案
    font_settings JSONB,            -- 字体设置
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- ============================================
-- PPT精修模块数据表
-- ============================================

-- 精修任务表
CREATE TABLE refinement_tasks (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    draft_id UUID REFERENCES assembly_drafts(id) ON DELETE SET NULL,
    title VARCHAR(255) NOT NULL,
    status VARCHAR(20) DEFAULT 'editing', -- editing, saved, exported
    total_pages INTEGER NOT NULL DEFAULT 0,
    total_modifications INTEGER DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 精修页面表（存储每页的最新内容）
CREATE TABLE refined_pages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id UUID REFERENCES refinement_tasks(id) ON DELETE CASCADE,
    page_index INTEGER NOT NULL,              -- 页面序号（0-based）
    slide_id UUID REFERENCES slides(id) ON DELETE SET NULL,
    source_document_id UUID REFERENCES documents(id) ON DELETE SET NULL,
    source_page_number INTEGER,
    content JSONB NOT NULL,                   -- 页面内容（元素、样式等）
    version INTEGER DEFAULT 1,                -- 页面版本号
    modification_count INTEGER DEFAULT 0,     -- 修改次数
    thumbnail_path VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(task_id, page_index)
);

-- 精修对话记录表
CREATE TABLE refinement_conversations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id UUID REFERENCES refinement_tasks(id) ON DELETE CASCADE,
    page_index INTEGER NOT NULL,              -- 关联的页面序号
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 精修对话消息表
CREATE TABLE refinement_messages (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    conversation_id UUID REFERENCES refinement_conversations(id) ON DELETE CASCADE,
    role VARCHAR(20) NOT NULL,                -- user, assistant, system
    content TEXT NOT NULL,
    message_type VARCHAR(20) DEFAULT 'chat',  -- chat, modification, suggestion
    modification_id UUID,                     -- 关联的修改记录ID
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 页面修改记录表（用于撤销/重做）
CREATE TABLE page_modifications (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id UUID REFERENCES refinement_tasks(id) ON DELETE CASCADE,
    page_index INTEGER NOT NULL,
    modification_type VARCHAR(50) NOT NULL,   -- edit_text, replace_image, edit_table, etc.
    description VARCHAR(255),                 -- 修改描述
    element_id VARCHAR(100),                  -- 被修改的元素ID
    element_type VARCHAR(50),                 -- text, image, table, chart, shape
    before_state JSONB,                       -- 修改前状态
    after_state JSONB,                        -- 修改后状态
    page_version INTEGER NOT NULL,            -- 修改后的页面版本
    is_undone BOOLEAN DEFAULT FALSE,          -- 是否已被撤销
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 智能建议记录表
CREATE TABLE refinement_suggestions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    task_id UUID REFERENCES refinement_tasks(id) ON DELETE CASCADE,
    page_index INTEGER NOT NULL,
    suggestion_type VARCHAR(50) NOT NULL,     -- text_length, tone, layout, style
    description VARCHAR(500) NOT NULL,
    element_id VARCHAR(100),                  -- 关联的元素ID
    confidence DECIMAL(3,2),                  -- 置信度 0-1
    is_applied BOOLEAN DEFAULT FALSE,         -- 是否已应用
    applied_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 创建索引
CREATE INDEX idx_documents_user_id ON documents(user_id);
CREATE INDEX idx_documents_status ON documents(status);
CREATE INDEX idx_documents_category ON documents(category_id);
CREATE INDEX idx_slides_document_id ON slides(document_id);
CREATE INDEX idx_slides_content ON slides USING gin(to_tsvector('chinese', content_text));
CREATE INDEX idx_drafts_user_id ON assembly_drafts(user_id);
CREATE INDEX idx_drafts_status ON assembly_drafts(status);
CREATE INDEX idx_chapters_draft_id ON draft_chapters(draft_id);
CREATE INDEX idx_selected_pages_chapter ON chapter_selected_pages(chapter_id);
CREATE INDEX idx_operation_history_draft ON operation_history(draft_id, created_at);

-- 精修模块索引
CREATE INDEX idx_refinement_tasks_user ON refinement_tasks(user_id);
CREATE INDEX idx_refinement_tasks_draft ON refinement_tasks(draft_id);
CREATE INDEX idx_refinement_tasks_status ON refinement_tasks(status);
CREATE INDEX idx_refined_pages_task ON refined_pages(task_id);
CREATE INDEX idx_refined_pages_task_index ON refined_pages(task_id, page_index);
CREATE INDEX idx_refinement_conversations_task ON refinement_conversations(task_id);
CREATE INDEX idx_refinement_messages_conversation ON refinement_messages(conversation_id);
CREATE INDEX idx_page_modifications_task ON page_modifications(task_id, page_index);
CREATE INDEX idx_page_modifications_version ON page_modifications(task_id, page_index, page_version);
CREATE INDEX idx_refinement_suggestions_task ON refinement_suggestions(task_id, page_index);

-- ============================================
-- PPT智能生成模块数据表
-- ============================================

-- PPT风格模板表（扩展，支持智能生成）
CREATE TABLE ppt_templates (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    name VARCHAR(100) NOT NULL,
    category VARCHAR(50) NOT NULL,            -- business, technology, education, minimalist, creative, academic
    thumbnail_path VARCHAR(500),
    preview_images JSONB,                     -- 预览图列表
    color_scheme JSONB NOT NULL,              -- 配色方案 {primary, secondary, background, text, accent}
    font_scheme JSONB NOT NULL,               -- 字体方案 {titleFont, bodyFont, titleSize, bodySize}
    is_system BOOLEAN DEFAULT FALSE,          -- 是否系统预置模板
    is_custom BOOLEAN DEFAULT FALSE,          -- 是否用户自定义上传
    source_file_path VARCHAR(500),            -- 上传的原始PPT文件路径

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP

    template_id UUID REFERENCES ppt_templates(id) ON DELETE SET NULL,
    status VARCHAR(20) DEFAULT 'idle',        -- idle, searching, analyzing, generating, applying_style, completed, failed
    progress INTEGER DEFAULT 0,               -- 进度百分比 0-100

    content JSONB NOT NULL,                   -- 页面内容（标题、正文、要点等）
    layout_type VARCHAR(50),                  -- 布局类型
    source_references JSONB,                  -- 内容引用来源
    thumbnail_path VARCHAR(500),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    UNIQUE(task_id, page_index)
);

-- 生成历史记录表
CREATE TABLE generation_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

    task_id UUID REFERENCES generation_tasks(id) ON DELETE SET NULL,
    topic VARCHAR(500) NOT NULL,              -- 主题摘要
    page_count INTEGER NOT NULL,
    template_name VARCHAR(100),               -- 使用的模板名称
    status VARCHAR(20) NOT NULL,              -- completed, failed
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    completed_at TIMESTAMP
);
```

```python
# 后端配置示例
class LLMCostController:
    """LLM成本控制器"""
    
    def __init__(self):
        self.daily_limit = int(os.getenv('LLM_DAILY_LIMIT', 50000))
        self.cache_enabled = os.getenv('LLM_CACHE_ENABLED', 'true') == 'true'
        self.cache_ttl = 3600  # 缓存1小时
        
    async def call_with_cache(self, prompt: str, **kwargs):
        """带缓存的LLM调用"""
        if self.cache_enabled:
            cache_key = hashlib.md5(prompt.encode()).hexdigest()
            cached = await redis.get(f"llm:{cache_key}")
            if cached:
                return cached
        
        # 检查每日限额
        daily_usage = await redis.incr(f"llm:daily:{date.today()}")
        if daily_usage > self.daily_limit:
            raise LLMQuotaExceeded("每日调用限额已达上限")
        
        result = await self.llm_client.chat(prompt, **kwargs)
        
        if self.cache_enabled:
            await redis.setex(f"llm:{cache_key}", self.cache_ttl, result)
        
        return result
```

```typescript
// 前端实现
class ChunkUploader {
  private chunkSize = 5 * 1024 * 1024; // 5MB

  async upload(file: File, onProgress: (progress: number) => void) {
    // 1. 初始化上传
    const totalChunks = Math.ceil(file.size / this.chunkSize);
    const initRes = await api.post('/documents/upload/init', {
      filename: file.name,
      filesize: file.size,
      total_chunks: totalChunks
    });
    const { upload_id } = initRes.data;

    // 2. 上传分片
    const uploadedChunks = [];
    for (let i = 0; i < totalChunks; i++) {
      const start = i * this.chunkSize;
      const end = Math.min(start + this.chunkSize, file.size);
      const chunk = file.slice(start, end);

      const formData = new FormData();
      formData.append('upload_id', upload_id);
      formData.append('chunk_index', i.toString());
      formData.append('chunk', chunk);

      await api.post('/documents/upload/chunk', formData, {
        headers: { 'Content-Type': 'multipart/form-data' }
      });

      uploadedChunks.push(i);
      onProgress((uploadedChunks.length / totalChunks) * 100);
    }

    // 3. 完成上传
    const completeRes = await api.post('/documents/upload/complete', {
      upload_id,
      title: file.name.replace(/\.[^/.]+$/, ''),
      category: ''
    });

    return completeRes.data;
  }
}
```

```python
# 后端实现
class UploadService:
    def __init__(self):
        self.temp_dir = Path("/app/data/temp")
        self.upload_dir = Path("/app/data/uploads")
  
    async def init_upload(self, filename: str, filesize: int, total_chunks: int) -> str:
        """初始化上传任务"""
        upload_id = str(uuid.uuid4())
        upload_info = {
            "upload_id": upload_id,
            "filename": filename,
            "filesize": filesize,
            "total_chunks": total_chunks,
            "received_chunks": [],
            "temp_dir": self.temp_dir / upload_id
        }
        # 保存到Redis
        await redis.setex(
            f"upload:{upload_id}",
            timedelta(hours=24),
            json.dumps(upload_info)
        )
        # 创建临时目录
        upload_info["temp_dir"].mkdir(parents=True, exist_ok=True)
        return upload_id
  
    async def upload_chunk(self, upload_id: str, chunk_index: int, chunk_data: bytes):
        """接收分片"""
        upload_info = json.loads(await redis.get(f"upload:{upload_id}"))
  
        # 保存分片
        chunk_path = upload_info["temp_dir"] / f"chunk_{chunk_index}"
        async with aiofiles.open(chunk_path, 'wb') as f:
            await f.write(chunk_data)
  
        # 更新进度
        upload_info["received_chunks"].append(chunk_index)
        await redis.setex(
            f"upload:{upload_id}",
            timedelta(hours=24),
            json.dumps(upload_info)
        )
  
    async def complete_upload(self, upload_id: str, title: str, category: str) -> Document:
        """合并分片，完成上传"""
        upload_info = json.loads(await redis.get(f"upload:{upload_id}"))
  
        # 合并文件
        final_path = self.upload_dir / f"{upload_id}_{upload_info['filename']}"
        async with aiofiles.open(final_path, 'wb') as outfile:
            for i in range(upload_info["total_chunks"]):
                chunk_path = upload_info["temp_dir"] / f"chunk_{i}"
                async with aiofiles.open(chunk_path, 'rb') as infile:
                    await outfile.write(await infile.read())
  
        # 清理临时文件
        shutil.rmtree(upload_info["temp_dir"])
        await redis.delete(f"upload:{upload_id}")
  
        # 创建文档记录
        document = await Document.create(
            title=title,
            file_path=str(final_path),
            file_size=upload_info["filesize"],
            status="parsing"
        )
  
        # 触发解析任务
        parse_ppt_task.delay(document.id)
  
        return document
```

```python
        slide_id: str
    ) -> str:
        """
        生成单页PPT缩略图
  
        使用 LibreOffice 将PPT转换为PDF，
        然后使用 Pillow 将指定页面转为图片
        """
        ppt_path = Path(ppt_path)
        output_dir = self.thumbnail_dir / slide_id[:2]
        output_dir.mkdir(parents=True, exist_ok=True)
  
        thumbnail_path = output_dir / f"{slide_id}.png"
  
        # 使用 LibreOffice 转换（只转换指定页面）
        # 注意：LibreOffice 不能直接转单页，需要转整个文件然后提取
        pdf_path = output_dir / f"{slide_id}.pdf"
  
        subprocess.run([
            "soffice",
            "--headless",
            "--convert-to", "pdf",
            "--outdir", str(output_dir),
            str(ppt_path)
        ], check=True, timeout=60)
  
        # 使用 pdf2image 转换指定页面
        from pdf2image import convert_from_path
  
        images = convert_from_path(
            pdf_path, 
            first_page=page_number,
            last_page=page_number,
            dpi=150  # 分辨率
        )
  
        if images:
            images[0].save(thumbnail_path, "PNG")
  
        # 清理临时PDF
        pdf_path.unlink(missing_ok=True)
  
        return str(thumbnail_path)
```

```python
# 权限检查装饰器
def require_owner(model_class, id_param="id"):
    """检查资源是否属于当前用户"""
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            current_user = kwargs.get("current_user")
            resource_id = kwargs.get(id_param)
  
            resource = await model_class.get_or_none(id=resource_id)
            if not resource:
                raise HTTPException(404, "Resource not found")
  
            if resource.user_id != current_user.id:
                raise HTTPException(403, "Permission denied")
  
            return await func(*args, **kwargs)
        return wrapper
    return decorator

# 使用示例
@app.delete("/api/documents/{document_id}")
@require_owner(Document, "document_id")
async def delete_document(document_id: str, current_user: User = Depends(get_current_user)):
    await Document.filter(id=document_id).delete()
    return {"success": True}
```
